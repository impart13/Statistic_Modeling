Вычисляем вероятность P(SP) попадания SP в интервал от SP- до SP+,¶

SP- = ( SP; SP) SP, SP+ = ( SP; SP) + SP

min()

max()

- np.mean

- np.std

- np.median

In [27]:
spmin = min (np.mean(sp), np.median(sp)) - np.std(sp) 
spmax = max (np.mean(sp), np.median(sp)) + np.std(sp)

kmin=0
kmax=0

for i in range(N-1):
    if sp[i] < spmin: 
        kmin += 1;
        if sp[i] > spmax:
            kmax += 1;

P = (2000-kmin-kmax)/2000

print ('SP- = ', spmin) 
print ('SP+ = ', spmax) 
print('P = ', P)
