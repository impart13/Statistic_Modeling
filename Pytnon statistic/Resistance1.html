Rmin = min (np.mean(R), np.median(R)) - np.std(R)
Rmax = max (np.mean(R), np.median(R)) + np.std(R)

kmin=0
kmax=0

for i in range(N-1):
    if R[i] < Rmin:
        kmin += 1;
    if R[i] > Rmax:
        kmax += 1;

P = (2000-kmin-kmax)/2000

print ('R- = ', Rmin)
print ('R+ = ', Rmax)
print ('kmin = ', kmin)
print ('kmax = ', kmax)
print('P = ', P)
